() => {
  const [sel, setSel] = useState(0)
  const items = useMemo(() => [
    'Item 1',
    'Item 2',
    'Item 3',
    'Item 4',
  ], [])

  return (
    <List style={{ width: 200 }}>
      {items.map((name, index) => (
        <ListItem
          button
          key={index}
          selected={sel === index}
          disabled={index === 2}
          onClick={() => setSel(index)}
        >
         {name}
        </ListItem>
      ))}
    </List>
  )
}
