() => {
  const bars = useMemo(() => {
    return new Array(60).fill(0).map(() => {
      let color = 'success'
      const rand = Math.floor(Math.random() * 100);
      if (rand < 5) color = ''
      else if (rand < 15) color = 'warning'
      else if (rand < 25) color = 'error'

      return {
        color,
        up: color === 'success'
      }
    })
  }, [])

  return (
    <Uptime
      label='Server status'
      showUptimePercent
      bars={bars}
      startDate='60 days ago'
      endDate='Today'
    />
  )
}
